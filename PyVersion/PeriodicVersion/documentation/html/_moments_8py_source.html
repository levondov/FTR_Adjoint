81 def CreateLatticeProfile(self, dB=[1.0], qstart=[0.0], qend=[0.1], qrot=[0.0], zstart=0, zend=1, repeat=1, verbose=False):
82
83 Nquads = len(dB) # number of quads
84 numCols = 4 # q start, q end, k strength, q rot
85
86 tmpLattice = np.zeros((1, numCols))
87
88 # drift from z start to q1 start
89 tmpLattice[0,:] = [zstart, qstart[0], 0.0, 0.0]
90
91 # loop through each quad and add to the lattice
92 for i in range(Nquads):
93
94 # add quad to lattice
95 tmpQuadPos = [[qstart[i], qend[i], dB[i], qrot[i]]]
96 tmpLattice = np.concatenate((tmpLattice, tmpQuadPos), axis=0)
97
98 # if this is not the last quad yet, add in the drift to the next quad
99 if (i+1) < Nquads:
100 tmpDrftPos = [[qend[i], qstart[i+1], 0.0, 0.0]]
101 tmpLattice = np.concatenate((tmpLattice, tmpDrftPos), axis=0)
102
103 # drfit from last q end to z end
104 tmpDrftPos = [[qend[-1], zend, 0.0, 0.0]]
105 tmpLattice = np.concatenate((tmpLattice, tmpDrftPos), axis=0)
106
107 # are we repeating the lattice?
108 if repeat > 1:
109 # make a copy of the lattice
110 copyTmpLattice = np.copy(tmpLattice)
111 for ii in range(repeat-1):
112 # grab the ending location of the last element
113 zendPos = tmpLattice[-1,1]
114
115 # create a copy of the lattice
116 newLatticePiece = np.copy(copyTmpLattice)
117
118 # adjust the starting and ending positions of this new piece
119 newLatticePiece[:,0:2] += zendPos
120
121 # append to the tmp lattce
122 tmpLattice = np.concatenate((tmpLattice, newLatticePiece), axis=0)
123
124
125 # assign to class parameter
126 self.latticelattice = tmpLattice
127 self.zstartzstart = zstart
128 self.zendzend = zend
129
130 if verbose:
131 print('Start | End | k strength | quad rotation')
132 print(self.latticelattice)

157 def __Get_beamridg_and_perv(self, energy=5e3, current=0.0):
158
159 # Parameters
160 e = 1.60217733E-19 #C
161 m = 9.1093897E-31 #kg
162 Energy = energy # eV
163 c = 2.997924E8 # m/s
164
165 gamma = 1+((Energy)/(510998.9461));
166 beta = np.sqrt((gamma*gamma)-1)/gamma
167 v = beta*c
168 bg = beta*gamma
169 rho = bg*c*(m/e)
170
171 k_perv = (1.0/(4.0*np.pi))*(c*377.0*current) / (m*v**3*gamma**3/e);
172
173 return rho,k_perv

178 def GetCOM(self, y):
179 L = y[9,:]
180 EQ = y[6,:]*y[0,:] + y[7,:]*y[1,:] + y[8,:]*y[2,:]
181 PP = y[3,:]**2 + y[4,:]**2 + y[5,:]**2
182 motion = EQ + (0.5)*L**2 - (0.5)*PP
183
184 return motion

191 def __ode3(self, F, verbose=False):
192
193 y0 = self.initialConditionsinitialConditions
194 h = self.hh
195
196 yout = np.array([]).reshape(len(y0),0)
197 tout = np.array([])
198 kval = np.array([])
199
200 # integrate through each element one at a time, making sure to match the initial conditions at the boundary points of elements
201 for j,elem in enumerate(self.latticelattice):
202
203 if verbose:
204 print("Integrating element: "+str(j+1))
205
206 # start, stop, and k value
207 t0 = elem[0]
208 t1 = elem[1]
209 k = elem[2]
210 tsteps = np.arange(t0,t1,h)
211 tout = np.concatenate((tout,tsteps))
212 kval = np.concatenate((kval,[k]*len(tsteps)))
213
214 # initialize output values for this element
215 ytmp = np.zeros((len(y0),len(tsteps)+1)) # +1 for initial value
216 # initial conditions are the very last set of points integrated in the previous element (except for the starting element)
217 ytmp[:,0] = yout[:,-1] if (j > 0) else y0
218
219 # run rk3 ode solver algorithm through the element
220 y = ytmp[:,0]
221 for i,t in enumerate(tsteps):
222 t1 = F(t,y,elem[2], elem[3])
223 s1 = h*t1
224 t2 = F(t+h/2.0, y+s1/2.0, elem[2], elem[3])
225 s2 = h*t2
226 t3 = F(t+h, y-s1+2.0*s2, elem[2], elem[3])
227 s3 = h*t3
228 y = y + (s1 + 4.0*s2 + s3)/6.0
229 ytmp[:,i+1] = y
230
231 # append to main output before moving onto next element
232 yout = np.concatenate((yout,ytmp),1) if (j==0) else np.concatenate((yout,ytmp[:,1:]),1)
233
234 tout = np.concatenate((tout,np.array([tout[-1]+h])))
235 kval = np.concatenate((kval,np.array([k])))
236 return tout,yout,kval

246 def __ode_moments(self, z, Y, dB, rot):
247 '''
248 Main function that solves Tom's moment equations
249
250 Y(1) - Q+
251 Y(2) - Q-
252 Y(3) - Qx
253 Y(4) - P+
254 Y(5) - P-
255 Y(6) - Px
256 Y(7) - E+
257 Y(8) - E-
258 Y(9) - Ex
259 Y(10) - L
260
261 '''
262
263 # quads
264 psi = rot
265 k_sol = 0.0
266 k_quad = dB / self.rigidityrigidity
267
268 # cq, sq quad calculations
269 cq = np.cos(2.0*Y[10]-2.0*psi)
270 sq = np.sin(2.0*Y[10]-2.0*psi)
271
272 # space charge calculation
273 Q_delta = np.sqrt( Y[0]**2 - Y[1]**2 - Y[2]**2 ) if Y[0]**2 - Y[1]**2 - Y[2]**2 >= 0 else 2**(-64.0)
274 ab4 = 1.0 / Q_delta; # the 4/ab term in equation
275 ca_ab4 = -Y[1] / ( (Y[0]+Q_delta)*Q_delta ) # 4c_alpha/ab
276 sa_ab4 = -Y[2] / ( (Y[0]+Q_delta)*Q_delta ) # 4s_alpha/ab
277
278 # calculate O,N matrix
279 O_mat,N_mat = self.__calcON__calcON(Y, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4)
280
281 # System of 10 equations
282 dydt = np.array([
283 # dQ/dz
284 Y[3],
285 Y[4],
286 Y[5],
287 # dP/dz
288 Y[6] + np.matmul( O_mat[0,:], np.reshape(Y[0:3],(3,1)) )[0],
289 Y[7] + np.matmul( O_mat[1,:], np.reshape(Y[0:3],(3,1)) )[0],
290 Y[8] + np.matmul( O_mat[2,:], np.reshape(Y[0:3],(3,1)) )[0],
291 # dE/dz
292 np.matmul( O_mat[0,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[0,0]*Y[9],
293 np.matmul( O_mat[1,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[1,0]*Y[9],
294 np.matmul( O_mat[2,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[2,0]*Y[9],
295 # dL/dz
296 -1.0*np.matmul( N_mat.T, Y[0:3] )[0],
297 # dphi/dz
298 -1.0*k_sol/2.0
299 ])
300
301 return dydt

306 def __calcON(self, Y, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4):
307 # pipe radius calculation
308 pipe_constant = (8*self.perveanceperveance/self.pipeRadiuspipeRadius**4) if self.pipeRadiuspipeRadius != 0 else 0 # zero pipe radius leads to infinity
309
310 # O and N matrix calculations
311 O_mat = np.array([[-k_sol**2/2.0 + ab4*self.perveanceperveance, 2.0*k_quad*cq + ca_ab4*self.perveanceperveance, -2.0*k_quad*sq + sa_ab4*self.perveanceperveance],
312 [2.0*k_quad*cq + ca_ab4*self.perveanceperveance, -k_sol**2/2.0 + ab4*self.perveanceperveance, 0],
313 [-2.0*k_quad*sq + sa_ab4*self.perveanceperveance, 0, -k_sol**2/2.0 + ab4*self.perveanceperveance]]) + pipe_constant*np.array([ [0, -Y[1], -Y[2]],[-Y[1], 0, 0],[-Y[2], 0, 0] ])
314
315 N_mat = np.array([[0], [2.0*k_quad*sq - sa_ab4*self.perveanceperveance], [2.0*k_quad*cq + ca_ab4*self.perveanceperveance]])
316 - pipe_constant*np.array([[0], [-Y[2]], [Y[1]]]) # pipe radius image forces addition
317
318 return O_mat,N_mat
