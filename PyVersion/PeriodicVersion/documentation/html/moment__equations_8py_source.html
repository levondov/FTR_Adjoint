7def run_moments(init_conditions, lattice, h, physics_params,verbose=False):
8 '''
9 INPUTS:
10 init_conditions - initial conditions for the integration
11 lattice - magnet profile (from CreateLatticeProfile()
12 h - integration step size
13 z_interval - integration interval
14 physics_params - list of physics params
15
16 OUTPUTS:
17 z - z position of solved ode (independent variable)
18 y - variables solved in ode
19 motion - constant of motion in the system
20 '''
21
22 # grab some more physics params
23 physics_params["rigidity"], physics_params["perveance"] = Get_beamridg_and_perv(physics_params["energy"],physics_params["current"])
24
25 odefunc = lambda z,Y,dbdx : ode_moments(z,Y,dbdx,physics_params)
26 z,y,k = ode3(odefunc, h, init_conditions, lattice, verbose_f=verbose)
27
28 # Constant of Motion %
29 motion = get_COM(y)
30
31 return z,y,motion,k

33def run_moments_adjoint(init_conditions, lattice, h, physics_params,verbose=False):

48def calcON(Y, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4, physics_params):
49 '''
50 Helper function that calculates the O and N matrix
51 '''
52 # pipe radius calculation
53 pipe_constant = (8*physics_params["perveance"]/physics_params["pipe_radius"]**4) if physics_params["pipe_radius"] != 0 else 0 # zero pipe radius leads to infinity
54
55 # O and N matrix calculations
56 O_mat = np.array([[-k_sol**2/2.0 + ab4*physics_params["perveance"], 2.0*k_quad*cq + ca_ab4*physics_params["perveance"], -2.0*k_quad*sq + sa_ab4*physics_params["perveance"]],
57 [2.0*k_quad*cq + ca_ab4*physics_params["perveance"], -k_sol**2/2.0 + ab4*physics_params["perveance"], 0],
58 [-2.0*k_quad*sq + sa_ab4*physics_params["perveance"], 0, -k_sol**2/2.0 + ab4*physics_params["perveance"]]]) + pipe_constant*np.array([ [0, -Y[1], -Y[2]],[-Y[1], 0, 0],[-Y[2], 0, 0] ])
59
60 N_mat = np.array([[0], [2.0*k_quad*sq - sa_ab4*physics_params["perveance"]], [2.0*k_quad*cq + ca_ab4*physics_params["perveance"]]])
61 - pipe_constant*np.array([[0], [-Y[2]], [Y[1]]]) # pipe radius image forces addition
62
63 return O_mat,N_mat

65def ode_moments(z,Y,quad_dbdx,physics_params):
66 '''
67 Main function that solves Tom's moment equations
68
69 Y(1) - Q+
70 Y(2) - Q-
71 Y(3) - Qx
72 Y(4) - P+
73 Y(5) - P-
74 Y(6) - Px
75 Y(7) - E+
76 Y(8) - E-
77 Y(9) - Ex
78 Y(10) - L
79
80 physics_params should have values for: rigidity, perveance, pipe_radius
81 '''
82
83 # quads
84 psi = 0.0
85 k_sol = 0.0
86 k_quad = quad_dbdx / physics_params["rigidity"]
87
88 # cq, sq quad calculations
89 cq = np.cos(2.0*Y[10]-2.0*psi)
90 sq = np.sin(2.0*Y[10]-2.0*psi)
91
92 # space charge calculation
93 Q_delta = np.sqrt( Y[0]**2 - Y[1]**2 - Y[2]**2 ) if Y[0]**2 - Y[1]**2 - Y[2]**2 >= 0 else 2**(-64.0)
94 ab4 = 1.0 / Q_delta; # the 4/ab term in equation
95 ca_ab4 = -Y[1] / ( (Y[0]+Q_delta)*Q_delta ) # 4c_alpha/ab
96 sa_ab4 = -Y[2] / ( (Y[0]+Q_delta)*Q_delta ) # 4s_alpha/ab
97
98 # calculate O,N matrix
99 O_mat,N_mat = calcON(Y, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4, physics_params)
100
101 # System of 10 equations
102 dydt = np.array([
103 # dQ/dz
104 Y[3],
105 Y[4],
106 Y[5],
107 # dP/dz
108 Y[6] + np.matmul( O_mat[0,:], np.reshape(Y[0:3],(3,1)) )[0],
109 Y[7] + np.matmul( O_mat[1,:], np.reshape(Y[0:3],(3,1)) )[0],
110 Y[8] + np.matmul( O_mat[2,:], np.reshape(Y[0:3],(3,1)) )[0],
111 # dE/dz
112 np.matmul( O_mat[0,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[0,0]*Y[9],
113 np.matmul( O_mat[1,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[1,0]*Y[9],
114 np.matmul( O_mat[2,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[2,0]*Y[9],
115 # dL/dz
116 -1.0*np.matmul( N_mat.T, Y[0:3] )[0],
117 # dphi/dz
118 -1.0*k_sol/2.0
119 ])
120
121 return dydt

123def ode_moments_adjoint(z,Yt,quad_dbdx,physics_params):
124 '''
125 Main function to solve the adjoint equations (backwards)
126
127 Here we have the original 11 moments + the 11 adjoint moments for a total of a 22 variable ode solve
128 '''
129 Y = Yt[0:11] # adjoint variables
130 Y2 = Yt[11:] # moment variables
131
132 # quads
133 psi = 0.0
134 k_sol = 0.0
135 k_quad = quad_dbdx / physics_params["rigidity"]
136
137 # calculations
138 cq = np.cos(2.0*Y2[10]-2.0*psi)
139 sq = np.sin(2.0*Y2[10]-2.0*psi)
140
141 # space charge stuff
142 Q_delta = np.sqrt( Y2[0]**2 - Y2[1]**2 - Y2[2]**2 ) if Y2[0]**2 - Y2[1]**2 - Y2[2]**2 >= 0 else 2**(-64.0)
143 ab4 = 1.0 / Q_delta; # the 4/ab term in equation
144 ca_ab4 = -Y2[1] / ( (Y2[0]+Q_delta)*Q_delta ) # 4c_alpha/ab
145 sa_ab4 = -Y2[2] / ( (Y2[0]+Q_delta)*Q_delta ) # 4s_alpha/ab
146
147 # calculate O,N matrix
148 O_mat,N_mat = calcON(Y2, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4, physics_params)
149
150 # Calculate special matrix due to space charge variations
151 [Mq,Mp,Mn] = get_SCVM(Y2,physics_params)
152
153 # System of 20 equations
154
155 # Solve regular envelope equations
156 dydt = np.array([
157 # dQ/dz
158 Y2[3],
159 Y2[4],
160 Y2[5],
161 # dP/dz
162 Y2[6] + np.matmul( O_mat[0,:], np.reshape(Y2[0:3],(3,1)) )[0],
163 Y2[7] + np.matmul( O_mat[1,:], np.reshape(Y2[0:3],(3,1)) )[0],
164 Y2[8] + np.matmul( O_mat[2,:], np.reshape(Y2[0:3],(3,1)) )[0],
165 # dE/dz
166 np.matmul( O_mat[0,:], np.reshape(Y2[3:6],(3,1)) )[0] + N_mat[0,0]*Y2[9],
167 np.matmul( O_mat[1,:], np.reshape(Y2[3:6],(3,1)) )[0] + N_mat[1,0]*Y2[9],
168 np.matmul( O_mat[2,:], np.reshape(Y2[3:6],(3,1)) )[0] + N_mat[2,0]*Y2[9],
169 # dL/dz
170 -1.0*np.matmul( N_mat.T, Y2[0:3] )[0],
171 # dphi/dz
172 -1.0*k_sol/2.0
173 ])
174
175 # dE dot 1 (y)
176 dedot1 = np.matmul( np.reshape(Y[3:6], (1,3)), Mq )[0] + \
177 Y[9]*np.matmul( np.reshape(Y2[0:3], (1,3)), Mn )[0] - \
178 np.matmul( np.reshape(Y[0:3], (1,3)), Mp )[0] - \
179 Y2[9]*np.matmul( np.reshape(Y[0:3], (1,3)), Mn )[0]
180 # dE dot 2 (y)
181 dedot2 = np.array([0,0,0])
182 # dE dot total (y)
183 dedot = dedot1 + dedot2
184 # dQ dot (y)
185 dqdot = np.array([0,0,0])
186 if z == 0.0:
187 dqdot = np.array([2*Y2[0],0,0])
188 # dP dot (y)
189 dpdot = np.array([0,0,0])
190 # dL dot (y)
191 dldot = 0
192
193 # Solve adjoint equations
194 dydt2 = np.array([
195 #dQ/dz (y)
196 Y[3] + dqdot[0],
197 Y[4] + dqdot[1],
198 Y[5] + dqdot[2],
199 # dP/dz (y)
200 Y[6] + np.matmul( O_mat[0,:], np.reshape(Y[0:3],(3,1)) )[0] + dpdot[0],
201 Y[7] + np.matmul( O_mat[1,:], np.reshape(Y[0:3],(3,1)) )[0] + dpdot[1],
202 Y[8] + np.matmul( O_mat[2,:], np.reshape(Y[0:3],(3,1)) )[0] + dpdot[2],
203 # dE/dz (y)
204 np.matmul( O_mat[0,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[0,0]*Y[9] + dedot[0],
205 np.matmul( O_mat[1,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[1,0]*Y[9] + dedot[1],
206 np.matmul( O_mat[2,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[2,0]*Y[9] + dedot[2],
207 # dL/dz
208 -1.0*np.matmul( N_mat.T, Y[0:3] )[0] + dldot,
209 # dphi/dz
210 -1.0*k_sol/2.0
211 ])
212
213 return np.concatenate((dydt2,dydt))

215def get_COM(y):
216 '''
217 Calculate the constant of motion
218 '''
219 L = y[9,:]
220 EQ = y[6,:]*y[0,:] + y[7,:]*y[1,:] + y[8,:]*y[2,:]
221 PP = y[3,:]**2 + y[4,:]**2 + y[5,:]**2
222 motion = EQ + (0.5)*L**2 - (0.5)*PP
223
224 return motion

227def get_SCVM(Y,physics_params):
228 '''
229 find space charge variation matrices needed for adjoint calculation
230 '''
231 k_perv = physics_params["perveance"]
232
233 Q_delta = np.sqrt( Y[0]**2 - Y[1]**2 - Y[2]**2 )
234 H = k_perv*np.log(Y[0] + Q_delta)
235 Q_deltaplus = Q_delta + Y[0]
236
237 V1 = -(k_perv/(Q_delta**2))*np.array([Y[3]-Y[1]*Y[4]/Q_deltaplus-Y[2]*Y[5]/Q_deltaplus,
238 -Y[1]*Y[3]/Q_deltaplus+Y[4],
239 -Y[2]*Y[3]/Q_deltaplus+Y[5]]).reshape(3,1)
240 U1t = (1./Q_delta)*np.array([Y[0], -Y[1], -Y[2]]).reshape(1,3)
241
242 V2 = (k_perv/(Q_delta*(Q_deltaplus**2)))*np.array([Y[1]*Y[4]+Y[2]*Y[5],Y[1]*Y[3],Y[2]*Y[3]]).reshape(3,1)
243 U2t = U1t + np.array([1.,0,0]).reshape(1,3)
244
245 V3 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[4], Y[3], 0]).reshape(3,1)
246 V4 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[5], 0, Y[3]]).reshape(3,1)
247 U3t = np.array([0, 1., 0]).reshape(1,3)
248 U4t = np.array([0,0,1.]).reshape(1,3)
249
250 Mp = V1*U1t + V2*U2t + V3*U3t + V4*U4t
251
252
253 W1 = -(k_perv/Q_delta)*np.array([1., Y[1]/Q_deltaplus, Y[2]/Q_deltaplus]).reshape(3,1)
254 W2 = (k_perv/(Q_delta*(Q_deltaplus**2)))*np.array([Y[1]**2+Y[2]**2,Y[1]*Y[0],Y[2]*Y[0]]).reshape(3,1)
255 W3 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[1],Y[0],0]).reshape(3,1)
256 W4 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[2],0,Y[0]]).reshape(3,1)
257
258 Mq = W1*U1t + W2*U2t + W3*U3t + W4*U4t
259
260
261 X1 = -k_perv*np.array([0, Y[2]/(Q_deltaplus*(Q_delta**2)), -Y[1]/(Q_deltaplus*(Q_delta**2))]).reshape(3,1)
262 X2 = -k_perv*np.array([0, Y[2]/(Q_delta*(Q_deltaplus**2)), -Y[1]/(Q_delta*(Q_deltaplus**2))]).reshape(3,1)
263 X3 = k_perv*np.array([0,0,-1./(Q_delta*Q_deltaplus)]).reshape(3,1)
264 X4 = k_perv*np.array([0,1./(Q_delta*Q_deltaplus),0]).reshape(3,1)
265
266 Mn = X1*U1t + X2*U2t + X3*U3t + X4*U4t
267
268 return Mq,Mp,Mn

270def get_ON_and_ACT(z,Y,Y_adj,k,physics_params):
271 '''
272 Calculate O and N matrices as well as the adjoint changing variables (dQ,dP,dE,dL with dots over them)
273 Note dEdot is actually dEdot_1 and dEdot_2, so 6 variables
274 '''
275
276 # grab some more physics params
277 physics_params["rigidity"], physics_params["perveance"] = Get_beamridg_and_perv(physics_params["energy"],physics_params["current"])
278
279 O = np.empty(len(z),dtype=object)
280 N = np.empty(len(z),dtype=object)
281 ACT = np.zeros((13,len(z)))
282
283 for i in range(len(z)):
284 # quads
285 psi = 0.0
286 k_sol = 0.0
287 k_quad = k[i] / physics_params["rigidity"]
288
289 # cq, sq quad calculations
290 cq = np.cos(2.0*Y[10,i]-2.0*psi)
291 sq = np.sin(2.0*Y[10,i]-2.0*psi)
292
293 # space charge calculation
294 Q_delta = np.sqrt( Y[0,i]**2 - Y[1,i]**2 - Y[2,i]**2 ) if Y[0,i]**2 - Y[1,i]**2 - Y[2,i]**2 >= 0 else 2**(-64.0)
295 ab4 = 1.0 / Q_delta; # the 4/ab term in equation
296 ca_ab4 = -Y[1,i] / ( (Y[0,i]+Q_delta)*Q_delta ) # 4c_alpha/ab
297 sa_ab4 = -Y[2,i] / ( (Y[0,i]+Q_delta)*Q_delta ) # 4s_alpha/ab
298
299 # calculate O,N matrix
300 O_mat,N_mat = calcON(Y[:,i], k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4, physics_params)
301
302 # Calculate special matrix due to space charge variations
303 [Mq,Mp,Mn] = get_SCVM(Y[:,i],physics_params)
304
305 # dE dot 1 (y)
306 dedot1 = np.matmul( np.reshape(Y_adj[3:6,i], (1,3)), Mq )[0] + \
307 Y_adj[9,i]*np.matmul( np.reshape(Y[0:3,i], (1,3)), Mn )[0] - \
308 np.matmul( np.reshape(Y_adj[0:3,i], (1,3)), Mp )[0] - \
309 Y[9,i]*np.matmul( np.reshape(Y_adj[0:3,i], (1,3)), Mn )[0]
310 # dE dot 2 (y)
311 dedot2 = np.array([0,0,0])
312 # dQ dot (y)
313 dqdot = np.array([0,0,0])
314 # dP dot (y)
315 dpdot = np.array([0,0,0])
316 # dL dot (y)
317 dldot = 0
318
319 ACT[0:3,i] = dqdot
320 ACT[3:6,i] = dpdot
321 ACT[6:9,i] = dedot1
322 ACT[9:12,i] = dedot2
323 ACT[12,i] = dldot
324 O[i] = O_mat
325 N[i] = N_mat
326
327 return O,N,ACT
