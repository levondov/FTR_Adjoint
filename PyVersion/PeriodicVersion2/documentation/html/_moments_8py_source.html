104 def CreateLatticeProfile(self, dB=[1.0], qstart=[0.0], qend=[0.1], qrot=[0.0], zstart=0, zend=1, repeat=1, verbose=False):
105
106 Nquads = len(dB) # number of quads
107 numCols = 4 # q start, q end, k strength, q rot
108
109 tmpLattice = np.zeros((1, numCols))
110
111 # drift from z start to q1 start
112 tmpLattice[0,:] = [zstart, qstart[0], 0.0, 0.0]
113
114 # loop through each quad and add to the lattice
115 for i in range(Nquads):
116
117 # add quad to lattice
118 tmpQuadPos = [[qstart[i], qend[i], dB[i], qrot[i]]]
119 tmpLattice = np.concatenate((tmpLattice, tmpQuadPos), axis=0)
120
121 # if this is not the last quad yet, add in the drift to the next quad
122 if (i+1) < Nquads:
123 tmpDrftPos = [[qend[i], qstart[i+1], 0.0, 0.0]]
124 tmpLattice = np.concatenate((tmpLattice, tmpDrftPos), axis=0)
125
126 # drfit from last q end to z end
127 tmpDrftPos = [[qend[-1], zend, 0.0, 0.0]]
128 tmpLattice = np.concatenate((tmpLattice, tmpDrftPos), axis=0)
129
130 # are we repeating the lattice?
131 if repeat > 1:
132 # make a copy of the lattice
133 copyTmpLattice = np.copy(tmpLattice)
134 for ii in range(repeat-1):
135 # grab the ending location of the last element
136 zendPos = tmpLattice[-1,1]
137
138 # create a copy of the lattice
139 newLatticePiece = np.copy(copyTmpLattice)
140
141 # adjust the starting and ending positions of this new piece
142 newLatticePiece[:,0:2] += zendPos
143
144 # append to the tmp lattce
145 tmpLattice = np.concatenate((tmpLattice, newLatticePiece), axis=0)
146
147
148 # create a reversed lattice for backwards integration
149 latticeReversed = np.copy(tmpLattice)
150 latticeReversed[:,0] = np.flip(tmpLattice[:,1])
151 latticeReversed[:,1] = np.flip(tmpLattice[:,0])
152 latticeReversed[:,2] = np.flip(tmpLattice[:,2])
153 latticeReversed[:,3] = np.flip(tmpLattice[:,3])
154
155 # assign to class parameter
156 self.latticelattice = tmpLattice
157 self.latticeReversedlatticeReversed = latticeReversed
158 self.zstartzstart = zstart
159 self.zendzend = zend
160
161 if verbose:
162 print('Start | End | k strength | quad rotation')
163 print(self.latticelattice)

234 def GetCOM(self, y):
235 L = y[9,:]

237 PP = y[3,:]**2 + y[4,:]**2 + y[5,:]**2
238 motion = EQ + (0.5)*L**2 - (0.5)*PP

251

253 if Y is None:
254 Y = self.yy
255
256 # new FoM matching condition
257 if 1:
258 # figure of merit broken into pieces for ease of reading

323 def GetAdjointIntegral(self, Omat, Nmat):
324 int1 = np.zeros(len(self.zz))
325 int2 = np.zeros(len(self.zz))
326 int3 = np.zeros(len(self.zz))
327 int4 = np.zeros(len(self.zz))
328
329 for i in range(len(self.zz)):
330 int1[i] = np.matmul(self.yAdjyAdj[3:6,i], np.matmul( Omat[0,i], self.yy[0:3,i] ) )
331 int2[i] = self.yAdjyAdj[9,i] * np.matmul( self.yy[0:3,i], Nmat[0,i] )[0]
332 int3[i] = -1*np.matmul(self.yAdjyAdj[0:3,i], np.matmul( Omat[0,i], self.yy[3:6,i] ) )
333 int4[i] = np.matmul(self.yAdjyAdj[0:3,i], Nmat[0,i] * self.yy[9,i] )
334
335 intVal = np.trapz(int1+int2+int3+int4, self.zz)
336
337 return intVal

343 def __Get_beamridg_and_perv(self, energy=5e3, current=0.0):
344
345 # Parameters
346 e = 1.60217733E-19 #C
347 m = 9.1093897E-31 #kg
348 Energy = energy # eV
349 c = 2.997924E8 # m/s
350
351 gamma = 1+((Energy)/(510998.9461));
352 beta = np.sqrt((gamma*gamma)-1)/gamma
353 v = beta*c
354 bg = beta*gamma
355 rho = bg*c*(m/e)
356
357 k_perv = (1.0/(4.0*np.pi))*(c*377.0*current) / (m*v**3*gamma**3/e);
358
359 return rho,k_perv

375 def __ode3(self, F, h, y0, lattice, verbose=False):
376
377 yout = np.array([]).reshape(len(y0),0)
378 tout = np.array([])
379 kval = np.array([])
380 Oval = np.array([], dtype=object).reshape(1,0)
381 Nval = np.array([], dtype=object).reshape(1,0)
382
383 if verbose:
384 print("Integrating elements ... ")
385
386 # integrate through each element one at a time, making sure to match the initial conditions at the boundary points of elements
387 for j,elem in enumerate(lattice):
388
389 if verbose:
390 print( str(j+1).zfill(3) + " / " + str(len(lattice[:,0])).zfill(3) )
391
392 # start, stop, and k value
393 t0 = elem[0]
394 t1 = elem[1]
395 k = elem[2]
396 tsteps = np.arange(t0,t1,h)
397 tout = np.concatenate((tout,tsteps))
398 kval = np.concatenate((kval,[k]*len(tsteps)))
399
400 # initialize output values for this element
401 ytmp = np.zeros((len(y0),len(tsteps)+1)) # +1 for initial value
402 Otmp = np.empty((1,len(tsteps)), dtype=object)
403 Ntmp = np.empty((1,len(tsteps)), dtype=object)
404
405 # initial conditions are the very last set of points integrated in the previous element (except for the starting element)
406 ytmp[:,0] = yout[:,-1] if (j > 0) else y0
407
408 # run rk3 ode solver algorithm through the element
409 y = ytmp[:,0]
410 for i,t in enumerate(tsteps):
411 t1, Otmp[0,i], Ntmp[0,i] = F(t,y,elem[2], elem[3])
412 s1 = h*t1
413 t2,_,_ = F(t+h/2.0, y+s1/2.0, elem[2], elem[3])
414 s2 = h*t2
415 t3,_,_ = F(t+h, y-s1+2.0*s2, elem[2], elem[3])
416 s3 = h*t3
417 y = y + (s1 + 4.0*s2 + s3)/6.0
418 ytmp[:,i+1] = y
419
420 # append to main output before moving onto next element
421 yout = np.concatenate((yout,ytmp),1) if (j==0) else np.concatenate((yout,ytmp[:,1:]),1)
422 Oval = np.concatenate((Oval,Otmp),axis=1)
423 Nval = np.concatenate((Nval,Ntmp),axis=1)
424
425 tout = np.concatenate((tout,np.array([tout[-1]+h])))
426 kval = np.concatenate((kval,np.array([k])))
427
428 # final eval for O matrices
429 _, Otmp, Ntmp = F(tout[-1], yout[:,-1], lattice[-1,2], lattice[-1,3])
430 Otmp2, Ntmp2 = np.empty((1,1),dtype=object), np.empty((1,1),dtype=object)
431 Otmp2[0,0], Ntmp2[0,0] = Otmp, Ntmp
432 Oval = np.concatenate((Oval,Otmp2),axis=1)
433 Nval = np.concatenate((Nval,Ntmp2),axis=1)
434
435 return tout,yout,kval,Oval,Nval

445 def __ode_moments(self, z, Y, dB, rot):
446 '''
447 Main function that solves Tom's moment equations
448
449 Y(1) - Q+
450 Y(2) - Q-
451 Y(3) - Qx
452 Y(4) - P+
453 Y(5) - P-
454 Y(6) - Px
455 Y(7) - E+
456 Y(8) - E-
457 Y(9) - Ex
458 Y(10) - L
459 Y(11) - phi (solenoid related variable)
460
461 '''
462
463 # quads
464 psi = rot
465 k_sol = 0.0
466 k_quad = dB / self.rigidityrigidity
467
468 # cq, sq quad calculations
469 cq = np.cos(2.0*Y[10]-2.0*psi)
470 sq = np.sin(2.0*Y[10]-2.0*psi)
471
472 # space charge calculation
473 Q_delta = np.sqrt( Y[0]**2 - Y[1]**2 - Y[2]**2 ) if Y[0]**2 - Y[1]**2 - Y[2]**2 > 0 else 2**(-64.0)
474 ab4 = 1.0 / Q_delta; # the 4/ab term in equation
475 ca_ab4 = -Y[1] / ( (Y[0]+Q_delta)*Q_delta ) # 4c_alpha/ab
476 sa_ab4 = -Y[2] / ( (Y[0]+Q_delta)*Q_delta ) # 4s_alpha/ab
477
478 # calculate O,N matrix
479 O_mat,N_mat = self.__calcON__calcON(Y, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4)
480
481 # System of 10 equations
482 dydt = np.array([
483 # dQ/dz
484 Y[3],
485 Y[4],
486 Y[5],
487 # dP/dz
488 Y[6] + np.matmul( O_mat[0,:], np.reshape(Y[0:3],(3,1)) )[0],
489 Y[7] + np.matmul( O_mat[1,:], np.reshape(Y[0:3],(3,1)) )[0],
490 Y[8] + np.matmul( O_mat[2,:], np.reshape(Y[0:3],(3,1)) )[0],
491 # dE/dz
492 np.matmul( O_mat[0,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[0,0]*Y[9],
493 np.matmul( O_mat[1,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[1,0]*Y[9],
494 np.matmul( O_mat[2,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[2,0]*Y[9],
495 # dL/dz
496 -1.0*np.matmul( N_mat.T, Y[0:3] )[0],
497 # dphi/dz
498 -1.0*k_sol/2.0
499 ])
500
501 return dydt, O_mat, N_mat

512 def __ode_moments_adjoint(self, z, Yt, dB, rot):
513
514 Y = Yt[0:11] # adjoint variables
515 Y2 = Yt[11:] # moment variables
516
517 # quads
518 psi = rot
519 k_sol = 0.0
520 k_quad = dB / self.rigidityrigidity
521
522 # calculations
523 cq = np.cos(2.0*Y2[10]-2.0*psi)
524 sq = np.sin(2.0*Y2[10]-2.0*psi)
525
526 # space charge stuff
527 Q_delta = np.sqrt( Y2[0]**2 - Y2[1]**2 - Y2[2]**2 ) if Y2[0]**2 - Y2[1]**2 - Y2[2]**2 > 0 else 2**(-64.0)
528 ab4 = 1.0 / Q_delta; # the 4/ab term in equation
529 ca_ab4 = -Y2[1] / ( (Y2[0]+Q_delta)*Q_delta ) # 4c_alpha/ab
530 sa_ab4 = -Y2[2] / ( (Y2[0]+Q_delta)*Q_delta ) # 4s_alpha/ab
531
532 # calculate O,N matrix
533 O_mat,N_mat = self.__calcON__calcON(Y2, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4)
534
535 # Calculate special matrix due to space charge variations
536 [Mq,Mp,Mn] = self.__GetSCVM__GetSCVM(Y2)
537
538 # System of 20 equations
539
540 # Solve regular envelope equations
541 dydt = np.array([
542 # dQ/dz
543 Y2[3],
544 Y2[4],
545 Y2[5],
546 # dP/dz
547 Y2[6] + np.matmul( O_mat[0,:], np.reshape(Y2[0:3],(3,1)) )[0],
548 Y2[7] + np.matmul( O_mat[1,:], np.reshape(Y2[0:3],(3,1)) )[0],
549 Y2[8] + np.matmul( O_mat[2,:], np.reshape(Y2[0:3],(3,1)) )[0],
550 # dE/dz
551 np.matmul( O_mat[0,:], np.reshape(Y2[3:6],(3,1)) )[0] + N_mat[0,0]*Y2[9],
552 np.matmul( O_mat[1,:], np.reshape(Y2[3:6],(3,1)) )[0] + N_mat[1,0]*Y2[9],
553 np.matmul( O_mat[2,:], np.reshape(Y2[3:6],(3,1)) )[0] + N_mat[2,0]*Y2[9],
554 # dL/dz
555 -1.0*np.matmul( N_mat.T, Y2[0:3] )[0],
556 # dphi/dz
557 -1.0*k_sol/2.0
558 ])
559
560 # dE dot 1 (y)
561 dedot1 = np.matmul( np.reshape(Y[3:6], (1,3)), Mq )[0] + \
562 Y[9]*np.matmul( np.reshape(Y2[0:3], (1,3)), Mn )[0] - \
563 np.matmul( np.reshape(Y[0:3], (1,3)), Mp )[0] - \
564 Y2[9]*np.matmul( np.reshape(Y[0:3], (1,3)), Mn )[0]
565 # dE dot 2 (y)
566 dedot2 = np.array([0,0,0])
567 # dE dot total (y)
568 dedot = dedot1 + dedot2
569 # dQ dot (y)
570 dqdot = np.array([0,0,0])
571 if z == 0.0:
572 dqdot = np.array([2*Y2[0],0,0])
573 # dP dot (y)
574 dpdot = np.array([0,0,0])
575 # dL dot (y)
576 dldot = 0
577
578 # Solve adjoint equations
579 dydt2 = np.array([
580 #dQ/dz (y)
581 Y[3] + dqdot[0],
582 Y[4] + dqdot[1],
583 Y[5] + dqdot[2],
584 # dP/dz (y)
585 Y[6] + np.matmul( O_mat[0,:], np.reshape(Y[0:3],(3,1)) )[0] + dpdot[0],
586 Y[7] + np.matmul( O_mat[1,:], np.reshape(Y[0:3],(3,1)) )[0] + dpdot[1],
587 Y[8] + np.matmul( O_mat[2,:], np.reshape(Y[0:3],(3,1)) )[0] + dpdot[2],
588 # dE/dz (y)
589 np.matmul( O_mat[0,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[0,0]*Y[9] + dedot[0],
590 np.matmul( O_mat[1,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[1,0]*Y[9] + dedot[1],
591 np.matmul( O_mat[2,:], np.reshape(Y[3:6],(3,1)) )[0] + N_mat[2,0]*Y[9] + dedot[2],
592 # dL/dz
593 -1.0*np.matmul( N_mat.T, Y[0:3] )[0] + dldot,
594 # dphi/dz
595 -1.0*k_sol/2.0
596 ])
597
598 return np.concatenate((dydt2,dydt)), O_mat, N_mat

603 def __calcON(self, Y, k_sol, k_quad, cq, sq, ab4, ca_ab4, sa_ab4):
604 # pipe radius calculation
605 pipe_constant = (8*self.perveanceperveance/self.pipeRadiuspipeRadius**4) if self.pipeRadiuspipeRadius != 0 else 0 # zero pipe radius leads to infinity
606
607 # O and N matrix calculations
608 O_mat = np.array([[-k_sol**2/2.0 + ab4*self.perveanceperveance, 2.0*k_quad*cq + ca_ab4*self.perveanceperveance, -2.0*k_quad*sq + sa_ab4*self.perveanceperveance],

613 - pipe_constant*np.array([[0], [-Y[2]], [Y[1]]]) # pipe radius image forces addition
614

659 def __GetSCVM(self, Y):
660 k_perv = self.perveanceperveance
661
662 Q_delta = np.sqrt( Y[0]**2 - Y[1]**2 - Y[2]**2 )
663 Q_deltaplus = Q_delta + Y[0]
664
665 V1 = -(k_perv/(Q_delta**2))*np.array([Y[3]-Y[1]*Y[4]/Q_deltaplus-Y[2]*Y[5]/Q_deltaplus,
666 -Y[1]*Y[3]/Q_deltaplus+Y[4],
667 -Y[2]*Y[3]/Q_deltaplus+Y[5]]).reshape(3,1)
668 U1t = (1./Q_delta)*np.array([Y[0], -Y[1], -Y[2]]).reshape(1,3)
669
670 V2 = (k_perv/(Q_delta*(Q_deltaplus**2)))*np.array([Y[1]*Y[4]+Y[2]*Y[5],Y[1]*Y[3],Y[2]*Y[3]]).reshape(3,1)
671 U2t = U1t + np.array([1.,0,0]).reshape(1,3)
672
673 V3 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[4], Y[3], 0]).reshape(3,1)
674 V4 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[5], 0, Y[3]]).reshape(3,1)
675 U3t = np.array([0, 1., 0]).reshape(1,3)
676 U4t = np.array([0,0,1.]).reshape(1,3)
677
678 Mp = V1*U1t + V2*U2t + V3*U3t + V4*U4t
679
680
681 W1 = -(k_perv/Q_delta)*np.array([1., Y[1]/Q_deltaplus, Y[2]/Q_deltaplus]).reshape(3,1)
682 W2 = (k_perv/(Q_delta*(Q_deltaplus**2)))*np.array([Y[1]**2+Y[2]**2,Y[1]*Y[0],Y[2]*Y[0]]).reshape(3,1)
683 W3 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[1],Y[0],0]).reshape(3,1)
684 W4 = -(k_perv/(Q_delta*Q_deltaplus))*np.array([Y[2],0,Y[0]]).reshape(3,1)
685
686 Mq = W1*U1t + W2*U2t + W3*U3t + W4*U4t
687
688
689 X1 = -k_perv*np.array([0, Y[2]/(Q_deltaplus*(Q_delta**2)), -Y[1]/(Q_deltaplus*(Q_delta**2))]).reshape(3,1)
690 X2 = -k_perv*np.array([0, Y[2]/(Q_delta*(Q_deltaplus**2)), -Y[1]/(Q_delta*(Q_deltaplus**2))]).reshape(3,1)
691 X3 = k_perv*np.array([0,0,-1./(Q_delta*Q_deltaplus)]).reshape(3,1)
692 X4 = k_perv*np.array([0,1./(Q_delta*Q_deltaplus),0]).reshape(3,1)
693
694 Mn = X1*U1t + X2*U2t + X3*U3t + X4*U4t
695
696 return Mq,Mp,Mn
